1. Nhóm Cắt Giảm & Xóa Bỏ (Scope Reduction)
   SQL
   -- =====================================================================
   -- 1. XÓA BỎ CÁC PHÂN HỆ VƯỢT SCOPE MVP
   -- =====================================================================

-- Xóa phân hệ Real-time Chat
DROP TABLE IF EXISTS chat_messages CASCADE;
DROP TABLE IF EXISTS chats CASCADE;

-- Xóa phân hệ Fraud Detection
DROP TABLE IF EXISTS fraud_records_and_labels CASCADE;
DROP TABLE IF EXISTS device_sessions CASCADE;
-- Xóa luôn bảng config của Fraud nếu có
DROP TABLE IF EXISTS fraud_rule_configs CASCADE;

-- Tạo bảng thay thế cho Chat: contact_messages (Hỗ trợ khách hàng đơn giản)
CREATE TABLE IF NOT EXISTS contact_messages (
message_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
user_id BIGINT REFERENCES users(user_id) ON DELETE CASCADE,
subject VARCHAR(200) NOT NULL,
content TEXT NOT NULL,
status VARCHAR(20) DEFAULT 'PENDING', -- PENDING, REPLIED, RESOLVED
created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- =====================================================================
-- NỚI RÀNG BUỘC VẬN CHUYỂN
-- =====================================================================
-- Đơn hàng mới tạo chưa thể có mã tracking ngay
ALTER TABLE orders ALTER COLUMN tracking_number DROP NOT NULL;
ALTER TABLE orders ALTER COLUMN tracking_number SET DEFAULT NULL;

2. Nhóm Bổ Sung Ràng Buộc & Bảng Cốt Lõi (Blockers)
   SQL
   -- =====================================================================
   -- 2. CHUẨN HÓA TOKENS & UNIQUE CONSTRAINTS
   -- =====================================================================

-- Thiết kế lại bảng tokens (Giữ lại duy nhất refresh_token, bỏ token_value thừa)
ALTER TABLE tokens
DROP COLUMN IF EXISTS token_value,
ADD COLUMN expires_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP;

CREATE INDEX idx_tokens_user ON tokens(user_id);
CREATE INDEX idx_tokens_refresh ON tokens(refresh_token);

-- Bổ sung Unique Constraints quan trọng
ALTER TABLE reviews
ADD CONSTRAINT uq_reviews_user_product_order UNIQUE (user_id, product_id, order_id);

ALTER TABLE cart_items
ADD CONSTRAINT uq_cart_items_cart_variant UNIQUE (cart_id, variant_id);

-- Ràng buộc User: Phải có password nếu không dùng Social Login
ALTER TABLE users
ADD CONSTRAINT chk_users_password
CHECK (password IS NOT NULL OR google_account_id IS NOT NULL OR facebook_account_id IS NOT NULL);

-- Bổ sung trường cho users
ALTER TABLE users
ADD COLUMN email_verified_at TIMESTAMPTZ NULL,
ADD COLUMN last_login_at TIMESTAMPTZ NULL;

-- =====================================================================
-- TẠO MỚI CÁC BẢNG THEO SRS V2 (PAYMENTS & NOTIFICATIONS)
-- =====================================================================

CREATE TABLE IF NOT EXISTS payments (
payment_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
order_id BIGINT NOT NULL REFERENCES orders(order_id) ON DELETE CASCADE,
payment_method SMALLINT NOT NULL, -- 0: COD, 1: VNPAY, 2: MOMO
amount NUMERIC(15,2) NOT NULL,
transaction_id VARCHAR(255) NULL, -- Mã giao dịch từ cổng thanh toán trả về
payment_status VARCHAR(50) DEFAULT 'PENDING' NOT NULL, -- PENDING, COMPLETED, FAILED, REFUNDED
paid_at TIMESTAMPTZ NULL,
created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS notifications (
notification_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
user_id BIGINT NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
title VARCHAR(200) NOT NULL,
content TEXT NOT NULL,
notification_type VARCHAR(50) NOT NULL, -- SYSTEM, PROMOTION, ORDER
reference_id BIGINT NULL, -- Chứa order_id hoặc promotion_id tùy type
is_read BOOLEAN DEFAULT FALSE,
created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_notifications_user_read ON notifications(user_id, is_read);

3. Nhóm Tối Ưu Cho AI & Hiển Thị (Denormalization & Indexing)
   SQL
   -- =====================================================================
   -- 3. DENORMALIZE BẢNG PRODUCTS (Phục vụ Filter & Content-based RecSys)
   -- =====================================================================

ALTER TABLE products
ADD COLUMN min_price NUMERIC(15,2) DEFAULT 0.00,
ADD COLUMN max_price NUMERIC(15,2) DEFAULT 0.00,
ADD COLUMN view_count INT DEFAULT 0,
ADD COLUMN sold_count INT DEFAULT 0,
ADD COLUMN rating_avg DECIMAL(3,2) DEFAULT 0.00,
ADD COLUMN rating_count INT DEFAULT 0,
ADD COLUMN popularity_score DECIMAL(10,4) DEFAULT 0.00; -- (VD: Score = 0.5 x views + 0.3 x sales + 0.2 x ratings)

-- (Gợi ý: Cần viết thêm 1 Trigger AFTER INSERT/UPDATE/DELETE trên bảng product_variants
-- để tự động tính lại MIN(price) và MAX(price) rồi update ngược về bảng products).

-- =====================================================================
-- TỐI ƯU & PARTITION BẢNG USER_BEHAVIORS
-- =====================================================================

-- Xóa bảng cũ và tạo lại dưới dạng Partition Table theo tháng (RANGE PARTITION)
DROP TABLE IF EXISTS user_behaviors CASCADE;

CREATE TABLE user_behaviors (
-- Bỏ GENERATED ALWAYS AS IDENTITY ở bảng cha vì partition table trên PG có cơ chế riêng
user_behavior_id BIGSERIAL,
user_id BIGINT NULL REFERENCES users(user_id) ON DELETE SET NULL,
session_id VARCHAR(100) NOT NULL,
product_id BIGINT NULL REFERENCES products(product_id) ON DELETE SET NULL,
category_id BIGINT NULL REFERENCES categories(category_id) ON DELETE SET NULL,
action_type VARCHAR(50) NOT NULL,
keyword VARCHAR(255) NULL,
context_data JSONB NULL,
variant_id BIGINT NULL,
created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
duration_seconds INT NULL, -- Số giây dừng lại trên sản phẩm
source VARCHAR(50) NULL; -- Nguồn click: 'SEARCH', 'HOME_PAGE', 'RECSYS_SECTION'
PRIMARY KEY (user_behavior_id, created_at) -- Partition key bắt buộc phải nằm trong PK
) PARTITION BY RANGE (created_at);

-- Tạo Index trên bảng cha (sẽ tự động áp dụng xuống các partition con)
CREATE INDEX idx_behavior_user_time ON user_behaviors(user_id, created_at DESC);
CREATE INDEX idx_behavior_product_time ON user_behaviors(product_id, created_at DESC);
CREATE INDEX idx_behavior_action_time ON user_behaviors(action_type, created_at DESC);

-- Tạo sẵn một vài partition cho các tháng giữa năm 2026
-- (Đồ án có thể thiết lập cronjob tự động tạo partition cho tháng tiếp theo)
CREATE TABLE user_behaviors_y2026m05 PARTITION OF user_behaviors
FOR VALUES FROM ('2026-05-01') TO ('2026-06-01');

CREATE TABLE user_behaviors_y2026m06 PARTITION OF user_behaviors
FOR VALUES FROM ('2026-06-01') TO ('2026-07-01');

CREATE TABLE user_behaviors_y2026m07 PARTITION OF user_behaviors
FOR VALUES FROM ('2026-07-01') TO ('2026-08-01');

ALTER TABLE product_similarities
ADD COLUMN similarity_type VARCHAR(50) DEFAULT 'CONTENT_BASED' NOT NULL;
-- Xóa PK cũ, cập nhật PK mới bao gồm cả type
ALTER TABLE product_similarities DROP CONSTRAINT product_similarities_pkey;
ALTER TABLE product_similarities ADD PRIMARY KEY (product_id, similar_product_id, similarity_type);

-- 4. BẢNG USER_RECOMMENDATIONS (User-based)
ALTER TABLE user_recommendations
ADD COLUMN algorithm VARCHAR(50) DEFAULT 'HYBRID' NOT NULL;
-- Đổi tên 'score' thành 'recommendation_score' cho rõ nghĩa
ALTER TABLE user_recommendations RENAME COLUMN score TO recommendation_score;
-- Cập nhật PK bao gồm thuật toán
ALTER TABLE user_recommendations DROP CONSTRAINT user_recommendations_pkey;
ALTER TABLE user_recommendations ADD PRIMARY KEY (user_id, product_id, algorithm);

4. Nhóm Chuẩn Hóa Dữ Liệu (Consistency)
   SQL
   -- =====================================================================
   -- 4. ĐỒNG BỘ KIỂU DỮ LIỆU TIME & CURRENCY
   -- =====================================================================

-- Chuyển kiểu DATE/TIMESTAMP sang TIMESTAMPTZ
ALTER TABLE products
ALTER COLUMN created_at TYPE TIMESTAMPTZ USING created_at AT TIME ZONE 'Asia/Ho_Chi_Minh';

ALTER TABLE orders
ALTER COLUMN order_date TYPE TIMESTAMPTZ USING order_date::TIMESTAMPTZ,
ALTER COLUMN shipping_date TYPE TIMESTAMPTZ USING shipping_date::TIMESTAMPTZ;

-- Đồng bộ toàn bộ dữ liệu tiền tệ về NUMERIC(15,2)
ALTER TABLE cart_items
ALTER COLUMN total_money TYPE NUMERIC(15,2);

ALTER TABLE order_details
ALTER COLUMN order_detail_price TYPE NUMERIC(15,2),
ALTER COLUMN total_money TYPE NUMERIC(15,2);

-- =====================================================================
-- BỔ SUNG CỘT & TRIGGER UPDATED_AT CHO CÁC BẢNG CỐT LÕI
-- =====================================================================

-- Thêm cột updated_at nếu chưa có
ALTER TABLE products ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP;
ALTER TABLE categories ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP;
ALTER TABLE orders ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP;
ALTER TABLE product_variants ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP;

-- Gắn Trigger gọi hàm trigger_set_timestamp() đã tạo từ trước
CREATE TRIGGER trg_products_updated_at
BEFORE UPDATE ON products
FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp();

CREATE TRIGGER trg_categories_updated_at
BEFORE UPDATE ON categories
FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp();

CREATE TRIGGER trg_orders_updated_at
BEFORE UPDATE ON orders
FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp();

CREATE TRIGGER trg_product_variants_updated_at
BEFORE UPDATE ON product_variants
FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp();

DROP TABLE IF EXISTS user_stats CASCADE;

CREATE TABLE IF NOT EXISTS user_login_logs (
log_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
user_id BIGINT REFERENCES users(user_id) ON DELETE CASCADE,
ip_address VARCHAR(45) NOT NULL,
user_agent TEXT,
device_id VARCHAR(255), -- Hash đơn giản từ frontend
login_status VARCHAR(20) NOT NULL, -- 'SUCCESS', 'FAILED'
created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- Index cực kỳ quan trọng để query sliding window (ví dụ: lấy log trong 24h qua)
CREATE INDEX idx_login_logs_user_time ON user_login_logs(user_id, created_at DESC);

CREATE TABLE IF NOT EXISTS risk_alerts (
alert_id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
user_id BIGINT REFERENCES users(user_id) ON DELETE CASCADE,
order_id BIGINT NULL REFERENCES orders(order_id) ON DELETE CASCADE,
rule_code VARCHAR(50) NOT NULL, -- VD: 'R1_MULTIPLE_DEVICES', 'R3_HIGH_AMOUNT'
risk_level VARCHAR(20) NOT NULL, -- 'LOW', 'MEDIUM', 'HIGH', 'CRITICAL'
description TEXT NOT NULL,
status VARCHAR(20) DEFAULT 'PENDING', -- 'PENDING', 'RESOLVED', 'FALSE_ALARM'
created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_risk_alerts_status ON risk_alerts(status);

CREATE TABLE IF NOT EXISTS risk_rule_configs (
rule_code VARCHAR(50) PRIMARY KEY,
rule_name VARCHAR(255) NOT NULL,
threshold_value NUMERIC(15,2) NOT NULL, -- Ngưỡng kích hoạt (số lần, hoặc số tiền)
time_window_minutes INT NULL, -- Khung thời gian xét duyệt (nếu có)
is_active BOOLEAN DEFAULT TRUE,
updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- Bổ sung trigger updated_at
CREATE TRIGGER trg_risk_rule_configs_updated_at
BEFORE UPDATE ON risk_rule_configs
FOR EACH ROW EXECUTE FUNCTION trigger_set_timestamp();
